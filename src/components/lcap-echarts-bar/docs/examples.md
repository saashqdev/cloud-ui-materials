### Basic Usage

```html
<lcap-echarts-bar
    xAxis="fakeXAxis"
    yAxis="Indicator 1, Indicator 2, Indicator 3"
    xAxisTitle="Dimension"
    yAxisTitle="Indicator"
    title="Title"
    theme="theme3"
    legendName="Mathematics, Swedish, English"
    :titleFontSize=16
    titleFontStyle="italic"
    :allowDownload=true
    :allowShowHint=true
    :allowShowLegend=true
    axisSplitLine="both"
    axisSplitLineType="solid"
    labelPosition="inside"
    undefinedToZero='empty'
    :showXAxisLine=true
    :showYAxisLine=true
    :showXAxisLabel=true
    :showYAxisLabel=true
    xAxisType="xBase"
    xAxisLabelRotate="0">
</lcap-echarts-bar>
```
