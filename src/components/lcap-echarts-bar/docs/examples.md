### Basic Usage

```html
<lcap-echarts-bar 
    xAxis="fakeXAxis" 
    yAxis="指标1，指标2，指标3"
    xAxisTitle="维度" 
    yAxisTitle="指标" 
    title="标题"
    theme="theme3"
    legendName="数学，语文，英语"
    :titleFontSize=16
    titleFontStyle="italic"
    :allowDownload=true
    :allowShowHint=true
    :allowShowLegend=true
    axisSplitLine="both"
    axisSplitLineType="solid"
    labelPosition="inside"
    undefinedToZero='empty'
    :showXAxisLine=true
    :showYAxisLine=true
    :showXAxisLabel=true
    :showYAxisLabel=true
    xAxisType="xBase"
    xAxisLabelRotate="0">
</lcap-echarts-bar>
```
