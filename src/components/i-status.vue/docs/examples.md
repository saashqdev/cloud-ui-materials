### Basic Usage

``` html
<u-linear-layout>
    <i-status name="success">Running</i-status>
    <i-status name="paused">Service Stopped</i-status>
    <i-status name="warning">Exception</i-status>
    <i-status name="waiting">Restarting</i-status>
    <i-status name="error">Update Failed</i-status>
    <i-status name="enabled">Enable</i-status>
    <i-status name="disabled">Disabled</i-status>
    <i-status name="unused">Unused</i-status>
    <i-status name="end">End</i-status>
    <i-status name="expired">Expired</i-status>
    <i-status name="locked">Locked</i-status>
    <i-status name="unlocked">Unlocked</i-status>
    <i-status name="binding">Binded</i-status>
    <i-status name="migrating">Migration</i-status>
    <i-status name="unpaid">Unpaid</i-status>
</u-linear-layout>
```
